class Instrument {
    public void play() {
        System.out.println("Instrument.play()");
    }
}

class Wind extends Instrument {
    public void play() {
        System.out.println("Wind.play()");
    }
}

public class Music {
    public static void tune(Instrument i) {
        i.draw();
    }

    public static void main(String[] args) {
        Wind w = new Wind();
        tune(w); // upcasting
    }
}//Output: Wind.play()
