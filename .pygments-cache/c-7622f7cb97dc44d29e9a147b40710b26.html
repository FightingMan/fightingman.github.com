#include <stdlib.h>
    #include <stdio.h>
    #include <arpa/inet.h>
    int main() {
        char s[] = "127.0.0.1";
        struct in_addr address;
        if ((inet_pton(AF_INET, s, &address)) != 1) {
            return -1;
        }

        printf("%i --- %i --- %i", address.s_addr,
                htonl(address.s_addr), INADDR_LOOPBACK);
        return 0;
    }
