#include &lt;arpa/inet.h&gt;
int main() {
    char s[] = "127.0.0.1";
    struct in_addr address;
    if ((inet_pton(AF_INET, s, &amp;address)) != 1) {
        return -1;
    }

    printf("%i --- %i --- %i", address.s_addr,
            htonl(address.s_addr), INADDR_LOOPBACK);
    return 0;
}
