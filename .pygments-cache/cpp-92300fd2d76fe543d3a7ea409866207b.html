    include <iostream>
    #include "server.h"
    using namespace std;

    int main(int argc, char* argv[]) {
        try {
            if (argc != 4)  {
                cerr << "Usage: ftsrv <address> <port> <doc_root>\n";
                return 1;
            }
            ftsrv::server s(argv[1], argv[2], argv[3]);
            s.run();
        } catch(exception& e) {
            cerr <<
                "exception:" <<e.what()<< "\n";
        }   

        return0;
    }
